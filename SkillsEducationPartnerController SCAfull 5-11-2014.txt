    public SkillsEducationPartnerController(){
        positionId = ApexPages.CurrentPage().getParameters().get('posId');
        recordType = ApexPages.CurrentPage().getParameters().get('RecordType');
        
        SkillsEducationWrapper = new SkillsEducationWrapper();

            retrievePosition();
            displayMsg();
            
            if(Position.Actual_End_Date__c == null){
                    chkEnd = false;
            }else{
                    chkEnd = true;
            }
            if(Position.Actual_Start_Date__c == null){
                chkStart = false;
            }
            else{
                chkStart = true;
            }
            
            RecordType = System.currentPagereference().getParameters().get('RecordType');
            step1 = System.currentPagereference().getParameters().get('Step');
            if(RecordType == 'Internship/Corps Position Request')
            {
               Step2 = '8';               
            }
            if(RecordType == 'Community Position Request')
            {
                Step2 = '7';
            }
            if(RecordType == 'Crew Position Request')
            {
                Step2 = '7';
            }
            if(RecordType == 'Leader Position Request')
            {
                Step2 = '7';
            }
            
            retriveSelectedSkillsAndEducation(SkillsEducationWrapper);
        }
    }
    
    // Retrieve list of all Createable Position fields
    public void retrievePosition(){
        String strFields = '';
        SObjectType sObj = Schema.getGlobalDescribe().get('Position__c');
        List<Schema.SObjectField> lstFields = sObj.getDescribe().fields.getMap().values();
        for(Schema.SobjectField so: lstFields){
            Schema.DescribeFieldResult dfr = so.getDescribe();
            strFields += dfr.getName() + ',';
        }
        System.debug('strFields: '+strFields);
        
        if(strFields.endsWith(',')){
            strFields = strFields.subString(0,strFields.lastIndexOf(','));
        }
        String strQuery = 'Select '+strFields+' From Position__c where Id =: positionId';
        List<Position__c> lstPosition = Database.query(strQuery);
        if(lstPosition.size()>0){
             position = lstPosition[0];
         }
         else {
            lstPosition = new  List<Position__c>();
            position = new Position__c();
            Error.LogError(Label.Partner_Portal_Position_Not_Found_Message); 
         }        
    }

        Pagereference pg;

    public void retriveSelectedSkillsAndEducation(SkillsEducationWrapper objSkillsEducationWrapper)
    {
        String skillsAndEdu;
        list<String> skillsAndEduList = new list<String>();
        set<String> skillsAndEduSet = new set<String>(); 
        
        //skillsAndEdu = position.SelectedSkillsAndEducation__c;
        /*
        if(skillsAndEdu != null)
            skillsAndEduList = skillsAndEdu.split(':');
        for(String skills : skillsAndEduList)
        {
            skillsAndEduSet.add(skills);
        }*/
        
        if(position.SelectedSkillsAndEducation__c == null){
            SkillsAndEdu = populateSelectedSkillsAndEducationInLongTextArea(position);
        }else if(position.SelectedSkillsAndEducation__c != null){
            SkillsAndEdu = position.SelectedSkillsAndEducation__c;
        }
        
        System.debug('SkillsAndEdu====>'+SkillsAndEdu);
        
        skillsAndEduList = skillsAndEdu.split(':');
        
        System.debug('skillsAndEduList====>'+skillsAndEduList);
        
        for(String skills : skillsAndEduList)
        {
            skillsAndEduSet.add(skills);
        }
                
        if(skillsAndEduSet.contains('Electrician__c'))
            objSkillsEducationWrapper.isElectrician = true;
        
        if(skillsAndEduSet.contains('Plumbing__c'))
            objSkillsEducationWrapper.isPlumbing = true;
    
        //if(skillsAndEduSet.contains('Mechanics__c'))
         //   objSkillsEducationWrapper.isMechanics = true;
        
        if(skillsAndEduSet.contains('Carpentry__c'))
            objSkillsEducationWrapper.isCarpentry = true;
    
        if(skillsAndEduSet.contains('Construction__c'))
